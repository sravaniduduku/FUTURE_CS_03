 API Security Risk Analysis

import requests
import json
from datetime import datetime

class APISecurityAnalyzer:

    def __init__(self, url):
        self.url = url
        self.report = []

    def add_result(self, title, status, details):
        self.report.append({
            "title": title,
            "status": status,
            "details": details
        })

    def check_https(self):
        if self.url.startswith("https://"):
            self.add_result(
                "HTTPS Security",
                "PASS",
                "Secure HTTPS connection detected."
            )
        else:
            self.add_result(
                "HTTPS Security",
                "FAIL",
                "API does not use HTTPS."
            )

    def check_response_headers(self):
        try:
            response = requests.get(self.url, timeout=10)

            headers = response.headers

            security_headers = [
                "Content-Security-Policy",
                "X-Frame-Options",
                "X-Content-Type-Options",
                "Strict-Transport-Security"
            ]

            for header in security_headers:
                if header in headers:
                    self.add_result(
                        header,
                        "PASS",
                        f"{header} found."
                    )
                else:
                    self.add_result(
                        header,
                        "WARNING",
                        f"{header} missing."
                    )

        except Exception as e:
            self.add_result(
                "Header Analysis",
                "ERROR",
                str(e)
            )

    def check_authentication(self):
        try:
            response = requests.get(self.url, timeout=10)

            if response.status_code == 401:
                self.add_result(
                    "Authentication",
                    "PASS",
                    "Authentication required."
                )
            else:
                self.add_result(
                    "Authentication",
                    "WARNING",
                    "No authentication required."
                )

        except Exception as e:
            self.add_result(
                "Authentication",
                "ERROR",
                str(e)
            )

    def check_information_disclosure(self):
        try:
            response = requests.get(self.url, timeout=10)

            content = response.text.lower()

            keywords = [
                "password",
                "secret",
                "token",
                "apikey",
                "database"
            ]

            found = []

            for word in keywords:
                if word in content:
                    found.append(word)

            if found:
                self.add_result(
                    "Information Disclosure",
                    "WARNING",
                    f"Sensitive words found: {', '.join(found)}"
                )
            else:
                self.add_result(
                    "Information Disclosure",
                    "PASS",
                    "No obvious sensitive information exposed."
                )

        except Exception as e:
            self.add_result(
                "Information Disclosure",
                "ERROR",
                str(e)
            )

    def check_status_code(self):
        try:
            response = requests.get(self.url, timeout=10)

            self.add_result(
                "HTTP Status",
                "INFO",
                f"Returned Status Code: {response.status_code}"
            )

        except Exception as e:
            self.add_result(
                "HTTP Status",
                "ERROR",
                str(e)
            )

    def generate_report(self):
        filename = "api_security_report.txt"

        with open(filename, "w", encoding="utf-8") as file:

            file.write("=" * 60 + "\n")
            file.write("API SECURITY RISK ANALYSIS REPORT\n")
            file.write("=" * 60 + "\n\n")

            file.write(f"Target API: {self.url}\n")
            file.write(f"Generated: {datetime.now()}\n\n")

            for item in self.report:
                file.write(f"Check: {item['title']}\n")
                file.write(f"Status: {item['status']}\n")
                file.write(f"Details: {item['details']}\n")
                file.write("-" * 50 + "\n")

        print(f"\nReport saved as: {filename}")

    def run(self):
        print("\nStarting API Security Analysis...\n")

        self.check_https()
        self.check_response_headers()
        self.check_authentication()
        self.check_information_disclosure()
        self.check_status_code()

        print("\nRESULTS")
        print("=" * 50)

        for item in self.report:
            print(f"\n{item['title']}")
            print(f"Status : {item['status']}")
            print(f"Details: {item['details']}")

        self.generate_report()


def main():

    print("=" * 60)
    print("API SECURITY RISK ANALYZER")
    print("=" * 60)

    api_url = input("\nEnter API URL: ").strip()

    analyzer = APISecurityAnalyzer(api_url)
    analyzer.run()


if __name__ == "__main__":
    main()
